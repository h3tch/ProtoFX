buffer buff_pos {
	usage staticDraw
	file ../../samples/cube.xml|pos
}

buffer buff_color {
	usage staticDraw
	file ../../samples/cube.xml|color
}

buffer buff_idx {
	usage staticDraw
	file ../../samples/cube.xml|index
}

vertinput in_vert {
	attr buff_pos float 4
	attr buff_color float 4
}

shader vert vs_simple {
	#version 440
	
	uniform mat4 g_view;
	uniform mat4 g_proj;
	
	layout(location = 0) in vec4 in_pos;
	layout(location = 1) in vec4 in_col;
	layout(location = 0) out vec4 vs_col;
	
	void main ()
	{
		gl_Position = g_proj * g_view * in_pos;
		vs_col = in_col;
	}
}

shader frag fs_simple {
	#version 440
	
	layout(location = 0) in vec4 gs_col;
	out vec4 color;
	
	void main ()
	{
		color = gs_col;
	}
}

pass pass_render {
	vert vs_simple
	frag fs_simple
	
	ClearColor 0.1 0.3 1.0 0.0
	Clear DepthBufferBit
	
	// draw <indexbuffername=0> <vertinname> <indextype> <primitivetype> <indexcount> <instancecount=0> <baseindex=0> <baseinstance=0> <basevertex=0>
	draw buff_idx in_vert unsignedshort triangles 36
}

tech tech_simple {
	pass pass_render
}
